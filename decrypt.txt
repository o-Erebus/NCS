#include <iostream>
#include <string>
using namespace std;

// Function to decrypt Vigenère Cipher
string vigenereDecrypt(const string& ciphertext, const string& keyword) {
    string plaintext = "";
    int keyLength = keyword.length();

    for (size_t i = 0; i < ciphertext.length(); ++i) {
        // Convert characters to 0–25 range
        char c = ciphertext[i];
        char k = keyword[i % keyLength];

        // Decryption formula: (C - K + 26) % 26
        char decryptedChar = ((c - 'A') - (k - 'A') + 26) % 26 + 'A';

        plaintext += decryptedChar;
    }

    return plaintext;
}

int main() {
    string ciphertext = "LXFOPVEFRNHR";
    string keyword = "LEMON";

    string decrypted = vigenereDecrypt(ciphertext, keyword);

    cout << "Ciphertext: " << ciphertext << endl;
    cout << "Keyword: " << keyword << endl;
    cout << "Decrypted Plaintext: " << decrypted << endl;

    return 0;
}
